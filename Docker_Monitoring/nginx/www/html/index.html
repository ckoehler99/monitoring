  • 02. Praktische Einführung

    Aufbau eines Monitoringsytems mit Docker und Überwachung von einzelnen Komponenten

Prometheus

docker-compose

prometheus:
  image: prom/prometheus
  container_name: prometheus
  volumes:
  - ../Docker_Monitoring_Data/prometheus:/prometheus
  - ./prometheus/etc:/etc
  command:
  - '-config.file=/etc/prometheus.yml'
  ports:
  - '9090:9090'
  networks:
  - monitoring

Grafana

docker-compose

grafana:
  image: grafana/grafana
  container_name: grafana
  environment:
  - GF_SECURITY_ADMIN_PASSWORD=pass
  volumes:
  - ./grafana/etc:/etc/grafana
  - ../Docker_Monitoring_Data/grafana/data:/var/lib/grafana
  depends_on:
  - prometheus
  ports:
   - "3000:3000"
   networks:
   - monitoring

Node Exporter

docker-compose

node-exporter:
  image: prom/node-exporter
  container_name: node-exporter
  ports:
  - '9100:9100'
  networks:
  - monitoring
